   function isci(zb1, zb2) {
    var i, j, k;
    for(i = 0; i <= zb1.length - zb2.length; i++) {
     for(j = i, k = 0; k < zb2.length; j++, k++) if(zb1[j] != zb2[k]) break;
     if(k == zb2.length) break;
    }
    if(i > zb1.length - zb2.length) i = -1;
    return i;
   }
  </script>
